366  private void TrapPlayerInRoom(bool boss) {
367  MVMain.Music.Play(boss ? "boss1" : MVMain.Room.activeRoomData.BGM_Name);
368  Tilemap ground = grid.transform.Find("Ground").GetComponent<Tilemap>();
369  if (boss) {
370  foreach (var tile in teleporter) {
371  if (tile.name[0] == 'I' && tile.name.Length == 2 && char.IsDigit(tile.name[1])) {
372  Vector3Int targetPos = tile.localPlace;
373  switch (tile.transitionMode) {
374  case TeleporterData.TeleporterType.Horizontal:
375  targetPos += (tile.localPlace.x < 0 ? Vector3Int.right : Vector3Int.left);
376  ground.SetTile(targetPos, trapperTile);
377  trapperTiles.Add(targetPos);
378  break;
379  case TeleporterData.TeleporterType.Vertical:
380  targetPos += (tile.localPlace.y < 0 ? Vector3Int.up : Vector3Int.down);
381  ground.SetTile(targetPos, trapperTile);
382  trapperTiles.Add(targetPos);
383  break;
384  default:
385  break;
386  }
387  }
388  }
389  }
390  else {
391  foreach (Vector3Int pos in trapperTiles) {
392  ground.SetTile(pos, null);
393  }
394  trapperTiles.Clear();
395  }
396  playerTrapped = boss;
397  }
